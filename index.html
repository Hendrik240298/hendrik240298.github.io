packages = [ "sympy", ] from sympy import solve, init_printing, Integral, sqrt, symbols from sympy.printing.mathml import MathMLPresentationPrinter from sympy.core.sympify import sympify from sympy.abc import x from js import document import random a = random.randint(-10,10) b= random.randint(-10,10) c = random.randint(-10,10) expression = a*x**2 + b*x + c solution = solve(expression, x, dict=False) print(solution) for i in range(len(solution)): print(f"x_{i+1}: {solution[i]}") s = MathMLPresentationPrinter() xml = s._print(sympify(expression)) s.apply_patch() pretty_xml = xml.toprettyxml() s.restore_patch() pretty_xml = pretty_xml[:8] + 'f(x)=' + pretty_xml[8:] pretty_xml = '' + pretty_xml + '' c = document.getElementById('outputformula') c.innerHTML = pretty_xml
